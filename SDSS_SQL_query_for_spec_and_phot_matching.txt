SELECT 
   s.specobjid, s.ra, s.dec, s.z,
   p.objID, p.ra, p.dec,
   dbo.fPhotoTypeN(p.type) as type,
   p.modelMag_u, p.modelMagErr_u,
   p.modelMag_g, p.modelMagErr_g,
   p.modelMag_r, p.modelMagErr_r, 
   p.modelMag_i, p.modelMagErr_i,
   p.modelMag_z, p.modelMagErr_z
FROM #upload u
      JOIN #x x ON x.up_id = u.up_id
      JOIN PhotoTag p ON p.objID = x.objID
      JOIN SpecObjAll s ON p.objID = s.bestObjID
ORDER by x.up_id
